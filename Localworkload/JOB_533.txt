select movie_keyword.movie_id, movie_link.movie_id, title.md5sum, link_type.link from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_link.linked_movie_id < 1394139 AND link_type.link >= 'spin off from' AND title.title >= 'Rahma' order by movie_link.movie_id ASC, movie_keyword.movie_id DESC, link_type.link DESC, title.md5sum ASC;
