select movie_keyword.keyword_id, title.title, title.series_years, movie_keyword.id, min(complete_cast.status_id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id where comp_cast_type.id < 1 AND title.episode_nr < 54 group by movie_keyword.keyword_id, title.series_years, movie_keyword.id, title.title having min(complete_cast.status_id) > 4 order by movie_keyword.id DESC;
