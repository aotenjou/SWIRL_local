select link_type.id, title.md5sum, title.id, movie_link.id from movie_info JOIN title ON title.id = movie_info.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where title.imdb_index = 'II' AND movie_info.movie_id >= 2059545 AND title.production_year < 1970 order by movie_link.id ASC;
