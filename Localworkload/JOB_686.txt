SELECT title.kind_id, comp_cast_type.id, complete_cast.subject_id, kind_type.kind FROM kind_type JOIN title ON title.kind_id = kind_type.id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id WHERE complete_cast.id >= 13834 AND complete_cast.status_id <= 4 AND title.episode_nr <= 54 AND title.imdb_index = 'I' AND title.episode_of_id <> 963153;
