select movie_keyword.movie_id, title.kind_id, title.md5sum, link_type.link from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where title.kind_id <= 1 AND movie_keyword.keyword_id = 12126 AND title.title >= '(#1.5035)' order by link_type.link ASC, movie_keyword.movie_id DESC, title.md5sum DESC, title.kind_id ASC;
