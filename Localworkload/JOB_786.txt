select info_type.id, company_name.name, movie_info_idx.movie_id, movie_companies.movie_id, movie_companies.company_type_id from info_type JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id JOIN title ON title.id = movie_info_idx.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_name ON company_name.id = movie_companies.company_id where title.id >= 2108925 AND title.episode_of_id >= 1079578 order by company_name.name ASC;
