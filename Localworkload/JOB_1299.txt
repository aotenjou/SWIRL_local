select complete_cast.subject_id, title.title, cast_info.note, char_name.id, sum(complete_cast.movie_id) from char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where cast_info.note = '(set location manager)' AND complete_cast.subject_id > 2 group by complete_cast.subject_id, title.title, char_name.id, cast_info.note having sum(complete_cast.movie_id) = 2427152;
