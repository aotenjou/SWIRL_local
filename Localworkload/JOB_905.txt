select title.episode_of_id, movie_info.info_type_id, movie_link.id, max(title.episode_nr) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.movie_id < 2143718 AND title.id > 563730 AND title.series_years >= '2009-????' group by title.episode_of_id, movie_info.info_type_id, movie_link.id having max(title.episode_nr) >= 29;
