SELECT title.phonetic_code, cast_info.role_id, char_name.imdb_index, movie_link.id FROM char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN movie_link ON movie_link.movie_id = title.id WHERE movie_link.linked_movie_id < 1394139 AND char_name.name > 'Adam First' AND title.id >= 2199088 ORDER BY cast_info.role_id ASC, movie_link.id DESC, char_name.imdb_index DESC;
