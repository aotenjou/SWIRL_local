select movie_keyword.movie_id, link_type.id, movie_keyword.keyword_id, movie_link.link_type_id from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where title.production_year < 1970 AND title.series_years >= '2009-????' AND title.title >= '(#1.5035)' order by link_type.id ASC, movie_keyword.movie_id DESC, movie_link.link_type_id DESC, movie_keyword.keyword_id ASC;
