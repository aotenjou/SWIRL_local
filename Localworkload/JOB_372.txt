select link_type.link, title.season_nr, movie_link.movie_id, avg(movie_link.link_type_id), max(title.phonetic_code) from link_type, movie_link, title where title.imdb_index <= 'I' AND title.id < 1827700 AND title.production_year = 1947 group by link_type.link, title.season_nr, movie_link.movie_id order by movie_link.movie_id DESC;
