SELECT title.title, complete_cast.movie_id, comp_cast_type.kind, MAX(title.title) FROM comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id WHERE comp_cast_type.id = 4 AND title.production_year < 1986 AND title.episode_nr <> 48 GROUP BY title.title, complete_cast.movie_id, comp_cast_type.kind HAVING MAX(title.title) > 'Fuyuki nontoruma no shisha de arimasu';
