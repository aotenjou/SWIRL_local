SELECT name.name_pcode_cf, person_info.info_type_id FROM name JOIN person_info ON person_info.person_id = name.id WHERE name.name_pcode_cf <= 'R2562' AND person_info.note <= 'Anonymous' AND name.id <= 3052160 AND name.imdb_index > 'VII' ORDER BY name.name_pcode_cf DESC, person_info.info_type_id ASC;
