SELECT title.season_nr, movie_link.link_type_id, link_type.id, movie_companies.note, MAX(title.episode_of_id) FROM link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id WHERE movie_link.id > 16299 AND movie_companies.note = '(1914) (USA) (theatrical)' GROUP BY title.season_nr, movie_link.link_type_id, link_type.id, movie_companies.note ORDER BY movie_companies.note DESC, title.season_nr DESC, movie_link.link_type_id ASC;
