select movie_info.movie_id, movie_link.id, title.series_years, max(title.episode_nr) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where title.id < 1397431 AND title.episode_of_id > 963153 AND movie_info.info_type_id >= 107 group by movie_link.id, movie_info.movie_id, title.series_years having max(title.episode_nr) >= 237;
