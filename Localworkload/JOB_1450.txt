SELECT cast_info.nr_order, name.surname_pcode, char_name.surname_pcode, COUNT(cast_info.movie_id) FROM char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN name ON name.id = cast_info.person_id WHERE cast_info.role_id >= 6 AND char_name.name_pcode_nf <> 'M3' GROUP BY cast_info.nr_order, name.surname_pcode, char_name.surname_pcode HAVING COUNT(cast_info.movie_id) = 684582 ORDER BY COUNT(cast_info.movie_id) ASC;
