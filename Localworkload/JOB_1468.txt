select title.md5sum, title.title, comp_cast_type.kind, movie_info.info, sum(title.kind_id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.info_type_id >= 43 AND movie_info.movie_id <= 2342737 AND complete_cast.status_id < 3 group by title.title, movie_info.info, comp_cast_type.kind, title.md5sum;
