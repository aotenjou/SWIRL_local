select cast_info.nr_order, title.production_year, char_name.id, complete_cast.subject_id, sum(complete_cast.movie_id) from char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where cast_info.note > '(executive producer) (as Robi Reed-Humes)' AND title.id > 443022 group by title.production_year, complete_cast.subject_id, char_name.id, cast_info.nr_order having sum(complete_cast.movie_id) = 2427152;
