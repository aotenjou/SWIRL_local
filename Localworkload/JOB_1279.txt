select movie_keyword.movie_id, movie_link.link_type_id, title.production_year, link_type.link from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_link.movie_id >= 132440 AND link_type.link >= 'spin off from' AND title.title >= '(#1.5035)' order by movie_keyword.movie_id ASC, movie_link.link_type_id DESC, link_type.link DESC, title.production_year ASC;
