select title.title, complete_cast.subject_id, title.md5sum, movie_info_idx.movie_id, info_type.info from movie_info JOIN info_type ON info_type.id = movie_info.info_type_id JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id JOIN title ON title.id = movie_info_idx.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where movie_info_idx.info_type_id = 99 AND movie_info.movie_id = 695564 order by complete_cast.subject_id DESC;
