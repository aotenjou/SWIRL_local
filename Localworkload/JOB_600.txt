select person_info.info, min(person_info.note), max(person_info.person_id) from person_info where person_info.info = '1952' AND person_info.person_id < 197068 AND person_info.info_type_id >= 24 group by person_info.info having min(person_info.note) = 'Anonymous' AND max(person_info.person_id) < 3048204 order by min(person_info.note) ASC, max(person_info.person_id) DESC;
