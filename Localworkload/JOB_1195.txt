SELECT title.episode_of_id, cast_info.movie_id, role_type.role FROM title JOIN cast_info ON cast_info.movie_id = title.id JOIN role_type ON role_type.id = cast_info.role_id WHERE cast_info.movie_id < 2369513 AND title.season_nr >= 13 AND title.md5sum >= '923f54faa9d41a1dac963b1e4ba1b44b' AND title.episode_of_id < 576415 ORDER BY title.episode_of_id ASC, cast_info.movie_id ASC;
