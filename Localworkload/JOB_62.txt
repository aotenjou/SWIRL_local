select kind_type.id, title.series_years, title.title, name.name_pcode_nf, cast_info.role_id from kind_type JOIN aka_title ON aka_title.kind_id = kind_type.id JOIN title ON title.id = aka_title.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN name ON name.id = cast_info.person_id where aka_title.episode_of_id < 14559 AND aka_title.season_nr = 6 order by cast_info.role_id DESC, kind_type.id DESC;
