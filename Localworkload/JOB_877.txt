SELECT movie_info_idx.id, AVG(movie_info_idx.movie_id) FROM movie_info_idx WHERE movie_info_idx.id <> 698866 AND movie_info_idx.movie_id > 1604324 AND movie_info_idx.info_type_id <> 100 GROUP BY movie_info_idx.id HAVING AVG(movie_info_idx.movie_id) < 1930390 ORDER BY AVG(movie_info_idx.movie_id) ASC;
