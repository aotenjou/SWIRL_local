SELECT movie_companies.id, title.phonetic_code, company_type.id, kind_type.kind, MIN(title.episode_of_id), MIN(title.md5sum) FROM company_type JOIN movie_companies ON movie_companies.company_type_id = company_type.id JOIN title ON title.id = movie_companies.movie_id JOIN kind_type ON kind_type.id = title.kind_id WHERE title.md5sum <> '4a9e824123c74028c8236f0cc3e13667' AND movie_companies.company_type_id >= 1 GROUP BY movie_companies.id, title.phonetic_code, company_type.id, kind_type.kind HAVING MIN(title.md5sum) = '438afc197bf5adfc49019f0496a3797e';
