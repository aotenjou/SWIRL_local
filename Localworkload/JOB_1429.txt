select title.season_nr, title.title, link_type.id, movie_companies.company_type_id, max(title.episode_of_id) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_link.id > 23398 AND movie_companies.note > '(1914) (USA) (theatrical)' group by title.season_nr, link_type.id, title.title, movie_companies.company_type_id order by title.title DESC, title.season_nr DESC, movie_companies.company_type_id ASC;
