select char_name.id, cast_info.person_role_id, cast_info.person_id from role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id where char_name.imdb_index >= 'IV' AND role_type.id >= 9 AND cast_info.person_id = 1430034 order by char_name.id DESC, cast_info.person_role_id ASC;
