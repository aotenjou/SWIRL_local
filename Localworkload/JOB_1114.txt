SELECT cast_info.person_role_id, name.surname_pcode, person_info.person_id, role_type.role, MAX(person_info.person_id) FROM person_info JOIN name ON name.id = person_info.person_id JOIN cast_info ON cast_info.person_id = name.id JOIN role_type ON role_type.id = cast_info.role_id WHERE name.name_pcode_cf > 'S3635' AND person_info.note < 'MilleniumBabyGirl' AND name.imdb_index > 'III' GROUP BY cast_info.person_role_id, name.surname_pcode, person_info.person_id, role_type.role;
