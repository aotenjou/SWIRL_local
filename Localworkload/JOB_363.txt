select title.md5sum, comp_cast_type.id, complete_cast.movie_id, title.episode_nr, min(complete_cast.status_id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id where comp_cast_type.id < 1 AND title.episode_nr < 54 group by title.episode_nr, comp_cast_type.id, complete_cast.movie_id, title.md5sum having min(complete_cast.status_id) > 3 order by title.md5sum DESC;
