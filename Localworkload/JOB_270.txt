select kind_type.id, title.season_nr, aka_title.production_year, aka_title.imdb_index from movie_companies JOIN title ON title.id = movie_companies.movie_id JOIN kind_type ON kind_type.id = title.kind_id JOIN aka_title ON aka_title.kind_id = kind_type.id where movie_companies.note <= '(2013) (Hong Kong) (DVD) (Blu-ray)' AND aka_title.imdb_index = 'XI' AND kind_type.kind >= 'episode';
