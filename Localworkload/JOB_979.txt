select cast_info.person_id, char_name.name, role_type.id from role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id where cast_info.note = '(voice) (as Steven Blum)' AND role_type.id <= 10 AND role_type.role = 'cinematographer' order by char_name.name DESC, cast_info.person_id DESC;
