select company_name.name, movie_info_idx.movie_id, info_type.id, title.id, title.episode_of_id from info_type JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id JOIN title ON title.id = movie_info_idx.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_name ON company_name.id = movie_companies.company_id where movie_companies.company_type_id >= 2 AND movie_companies.movie_id >= 1092736 order by company_name.name ASC;
