SELECT name.surname_pcode, movie_info_idx.movie_id, cast_info.note, info_type.id, role_type.id, person_info.note FROM movie_info_idx JOIN info_type ON info_type.id = movie_info_idx.info_type_id JOIN person_info ON person_info.info_type_id = info_type.id JOIN name ON name.id = person_info.person_id JOIN cast_info ON cast_info.person_id = name.id JOIN role_type ON role_type.id = cast_info.role_id WHERE role_type.id = 9 AND name.name_pcode_cf >= 'H6252';
