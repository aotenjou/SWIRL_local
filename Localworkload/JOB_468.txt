select movie_keyword.movie_id, title.episode_nr, title.production_year, movie_link.linked_movie_id from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_link.linked_movie_id >= 1808225 AND link_type.link >= 'spin off from' AND title.title >= '(#1.5035)' order by movie_keyword.movie_id ASC, title.episode_nr DESC, movie_link.linked_movie_id DESC, title.production_year ASC;
