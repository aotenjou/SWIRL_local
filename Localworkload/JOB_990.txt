select cast_info.note, char_name.name, char_name.id, cast_info.person_role_id, sum(complete_cast.movie_id) from char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where complete_cast.status_id <= 4 AND complete_cast.subject_id > 2 group by cast_info.note, cast_info.person_role_id, char_name.id, char_name.name having sum(complete_cast.movie_id) = 2427152;
