SELECT keyword.phonetic_code, movie_keyword.movie_id, MIN(keyword.phonetic_code), AVG(keyword.id), SUM(movie_keyword.id) FROM movie_keyword JOIN keyword ON keyword.id = movie_keyword.keyword_id WHERE keyword.id >= 5165 AND movie_keyword.keyword_id > 15722 AND movie_keyword.id <> 205842 GROUP BY keyword.phonetic_code, movie_keyword.movie_id HAVING MIN(keyword.phonetic_code) < 'C4265' ORDER BY movie_keyword.movie_id DESC;
