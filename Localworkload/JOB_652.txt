select movie_info.movie_id, title.season_nr, title.episode_of_id, movie_link.id from movie_info JOIN title ON title.id = movie_info.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_link.movie_id = 8087 AND title.id >= 1103840 AND title.production_year < 1970 order by movie_info.movie_id ASC;
