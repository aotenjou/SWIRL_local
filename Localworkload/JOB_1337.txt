select complete_cast.movie_id, keyword.id, keyword.keyword, movie_keyword.id, comp_cast_type.kind from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id where title.id >= 386873 AND keyword.phonetic_code <= 'P4352' AND complete_cast.subject_id >= 1 order by movie_keyword.id DESC, keyword.keyword DESC;
