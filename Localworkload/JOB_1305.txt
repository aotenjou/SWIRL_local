SELECT movie_companies.id, company_type.kind, MIN(company_type.kind), MIN(company_type.id) FROM movie_companies, company_type WHERE movie_companies.company_type_id = 1 AND movie_companies.company_id <= 34 AND movie_companies.id > 1098778 GROUP BY movie_companies.id, company_type.kind HAVING MIN(company_type.kind) < 'miscellaneous companies' ORDER BY company_type.kind ASC, movie_companies.id DESC;
