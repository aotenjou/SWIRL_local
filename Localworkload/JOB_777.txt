SELECT person_info.info_type_id, info_type.id, movie_info_idx.info_type_id FROM person_info JOIN info_type ON info_type.id = person_info.info_type_id JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id WHERE movie_info_idx.id <= 698866 AND person_info.note >= 'Anonymous' AND person_info.info_type_id > 29 ORDER BY info_type.id DESC, person_info.info_type_id ASC, movie_info_idx.info_type_id DESC;
