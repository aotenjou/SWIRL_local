SELECT title.production_year, movie_info_idx.info_type_id, movie_companies.company_type_id FROM movie_companies JOIN title ON title.id = movie_companies.movie_id JOIN movie_info_idx ON movie_info_idx.movie_id = title.id WHERE title.series_years <> '2009-????' AND movie_info_idx.id <> 1096919 AND title.production_year <> 1902 AND title.title > 'The Redneck Yacht Club' ORDER BY movie_info_idx.info_type_id ASC, movie_companies.company_type_id ASC;
