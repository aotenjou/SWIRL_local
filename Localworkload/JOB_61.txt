select movie_companies.note, movie_companies.company_type_id, count(movie_companies.company_type_id), sum(movie_companies.movie_id), max(movie_companies.id) from company_type, movie_companies where movie_companies.movie_id < 1761913 AND company_type.kind >= 'special effects companies' AND company_type.kind = 'special effects companies' group by movie_companies.company_type_id, movie_companies.note;
