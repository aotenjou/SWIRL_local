SELECT movie_companies.company_type_id, title.title, movie_info.note, company_type.id FROM movie_info JOIN title ON title.id = movie_info.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE title.kind_id <= 1 AND movie_companies.id >= 1774586 AND title.id < 724807 AND movie_info.movie_id <= 2143718 AND company_type.kind >= 'special effects companies' AND title.production_year = 1902;
