SELECT cast_info.movie_id, name.name, person_info.info, info_type.id, role_type.id FROM info_type JOIN person_info ON person_info.info_type_id = info_type.id JOIN name ON name.id = person_info.person_id JOIN cast_info ON cast_info.person_id = name.id JOIN role_type ON role_type.id = cast_info.role_id WHERE role_type.id = 9 AND person_info.person_id <> 1904541 ORDER BY person_info.info ASC, info_type.id DESC;
