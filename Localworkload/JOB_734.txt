select company_name.id, title.episode_of_id, info_type.id, movie_info_idx.movie_id, movie_companies.id from info_type JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id JOIN title ON title.id = movie_info_idx.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_name ON company_name.id = movie_companies.company_id where title.season_nr <= 5 AND movie_companies.company_type_id >= 1 order by movie_info_idx.movie_id ASC;
