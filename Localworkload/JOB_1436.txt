select title.title, complete_cast.status_id, movie_info.movie_id, comp_cast_type.id, sum(comp_cast_type.id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_info ON movie_info.movie_id = title.id where title.season_nr >= 3 AND movie_info.movie_id <= 2342737 AND title.episode_of_id < 314959 group by title.title, comp_cast_type.id, movie_info.movie_id, complete_cast.status_id;
