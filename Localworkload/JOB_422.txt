select movie_keyword.movie_id, keyword.phonetic_code from keyword, movie_keyword where movie_keyword.keyword_id >= 75066 AND keyword.id = 86959 AND movie_keyword.movie_id >= 1845172 AND movie_keyword.id > 525101 AND keyword.phonetic_code < 'N6253' order by movie_keyword.movie_id DESC, keyword.phonetic_code DESC;
