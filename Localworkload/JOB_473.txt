SELECT cast_info.movie_id, title.series_years, movie_link.id, SUM(movie_link.linked_movie_id) FROM cast_info JOIN title ON title.id = cast_info.movie_id JOIN movie_link ON movie_link.movie_id = title.id WHERE cast_info.note <= '(written for television by)' AND title.kind_id > 2 GROUP BY cast_info.movie_id, title.series_years, movie_link.id HAVING SUM(movie_link.linked_movie_id) < 1330291 ORDER BY cast_info.movie_id DESC;
