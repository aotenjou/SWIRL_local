SELECT movie_info.info_type_id, title.kind_id, complete_cast.id, info_type.id, MIN(complete_cast.id) FROM info_type JOIN movie_info ON movie_info.info_type_id = info_type.id JOIN title ON title.id = movie_info.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id WHERE info_type.id < 59 AND complete_cast.movie_id > 1923775 GROUP BY movie_info.info_type_id, title.kind_id, complete_cast.id, info_type.id ORDER BY movie_info.info_type_id DESC, title.kind_id DESC, info_type.id DESC;
