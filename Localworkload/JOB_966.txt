select kind_type.id, movie_companies.movie_id, aka_title.title, title.imdb_index from movie_companies JOIN title ON title.id = movie_companies.movie_id JOIN kind_type ON kind_type.id = title.kind_id JOIN aka_title ON aka_title.kind_id = kind_type.id where aka_title.imdb_index <= 'XI' AND aka_title.imdb_index = 'XI' AND aka_title.production_year >= 1949;
