SELECT info_type.id, title.episode_of_id, complete_cast.id, movie_info_idx.movie_id, comp_cast_type.kind, movie_info.note FROM comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_info_idx ON movie_info_idx.movie_id = title.id JOIN info_type ON info_type.id = movie_info_idx.info_type_id JOIN movie_info ON movie_info.info_type_id = info_type.id WHERE title.imdb_index <> 'II' AND movie_info_idx.info_type_id <> 101;
