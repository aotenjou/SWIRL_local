SELECT movie_info.movie_id, title.season_nr, company_name.md5sum, movie_companies.note FROM company_name JOIN movie_companies ON movie_companies.company_id = company_name.id JOIN title ON title.id = movie_companies.movie_id JOIN movie_info ON movie_info.movie_id = title.id WHERE movie_info.info > 'MET:600 m' AND movie_info.note < '(French Film Festival)' AND movie_companies.company_id > 3351;
