SELECT aka_name.name_pcode_cf, name.name_pcode_nf, title.episode_of_id, cast_info.movie_id, movie_link.movie_id FROM aka_name JOIN name ON name.id = aka_name.person_id JOIN cast_info ON cast_info.person_id = name.id JOIN title ON title.id = cast_info.movie_id JOIN movie_link ON movie_link.movie_id = title.id WHERE title.production_year >= 2002 AND aka_name.surname_pcode = 'M2' AND aka_name.name <= 'Williams, Samiel';
