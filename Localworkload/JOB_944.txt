SELECT cast_info.id, aka_name.id, name.surname_pcode, COUNT(cast_info.person_id), MIN(cast_info.id) FROM aka_name JOIN name ON name.id = aka_name.person_id JOIN cast_info ON cast_info.person_id = name.id WHERE cast_info.movie_id < 95986 AND name.name <= 'Ibo Alfaro, Manuel' GROUP BY cast_info.id, aka_name.id, name.surname_pcode HAVING COUNT(cast_info.person_id) <= 2576539 AND MIN(cast_info.id) <> 20785264;
