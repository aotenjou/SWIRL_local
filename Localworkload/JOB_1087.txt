SELECT title.episode_nr, movie_keyword.keyword_id, aka_title.id, keyword.keyword, kind_type.kind FROM kind_type JOIN aka_title ON aka_title.kind_id = kind_type.id JOIN title ON title.id = aka_title.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id JOIN keyword ON keyword.id = movie_keyword.keyword_id WHERE kind_type.id <= 4 AND aka_title.episode_nr >= 424 ORDER BY kind_type.kind DESC;
