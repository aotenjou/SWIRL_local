select title.season_nr, movie_link.id, link_type.id, title.imdb_index from movie_info JOIN title ON title.id = movie_info.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_link.linked_movie_id >= 1394139 AND title.id >= 386873 AND title.production_year < 1970 order by title.imdb_index ASC;
