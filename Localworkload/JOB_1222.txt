SELECT title.id, movie_link.linked_movie_id, movie_keyword.keyword_id, keyword.keyword, SUM(keyword.id) FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id WHERE title.episode_of_id <= 1325926 AND movie_link.linked_movie_id >= 1238808 GROUP BY title.id, movie_link.linked_movie_id, movie_keyword.keyword_id, keyword.keyword ORDER BY movie_keyword.keyword_id DESC;
