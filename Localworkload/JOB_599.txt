select cast_info.person_role_id, name.md5sum, aka_name.imdb_index, role_type.id from aka_name JOIN name ON name.id = aka_name.person_id JOIN cast_info ON cast_info.person_id = name.id JOIN role_type ON role_type.id = cast_info.role_id where aka_name.imdb_index > 'I' AND cast_info.person_id <= 302472 AND cast_info.nr_order <= 69 AND role_type.id >= 9;
