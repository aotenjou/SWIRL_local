SELECT char_name.surname_pcode, cast_info.role_id, title.phonetic_code FROM char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id WHERE char_name.name_pcode_nf < 'R2636' AND title.season_nr <= 13 AND cast_info.nr_order <> 9 AND title.imdb_index = 'I';
