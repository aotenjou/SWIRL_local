SELECT movie_companies.company_type_id, title.series_years, movie_info.id, company_type.kind, MIN(company_type.kind) FROM movie_info JOIN title ON title.id = movie_info.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE title.title = '(#6.59)' AND title.production_year < 1986 GROUP BY movie_companies.company_type_id, title.series_years, movie_info.id, company_type.kind HAVING MIN(company_type.kind) = 'production companies';
