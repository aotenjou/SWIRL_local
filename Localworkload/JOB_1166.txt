select title.id, title.kind_id, title.episode_nr, keyword.id, title.production_year from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id where keyword.id >= 104859 AND keyword.keyword <= 'civic-pride' AND complete_cast.movie_id >= 1438895 order by title.production_year DESC, title.episode_nr DESC;
