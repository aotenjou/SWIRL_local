select movie_companies.company_type_id, movie_link.linked_movie_id, movie_link.movie_id, movie_companies.note, max(title.episode_of_id) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_link.id > 16299 AND movie_companies.note = '(1914) (USA) (theatrical)' group by movie_link.movie_id, movie_companies.company_type_id, movie_link.linked_movie_id, movie_companies.note order by movie_companies.note DESC, movie_link.movie_id DESC, movie_companies.company_type_id ASC;
