SELECT title.season_nr, movie_info_idx.info_type_id, cast_info.note, char_name.name_pcode_nf, AVG(cast_info.id) FROM movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN char_name ON char_name.id = cast_info.person_role_id WHERE movie_info_idx.info_type_id <> 101 AND cast_info.nr_order = 5 GROUP BY title.season_nr, movie_info_idx.info_type_id, cast_info.note, char_name.name_pcode_nf HAVING AVG(cast_info.id) <= 3774382 ORDER BY AVG(cast_info.id) ASC;
