SELECT cast_info.id, aka_name.name_pcode_nf, char_name.name, name.name_pcode_nf, COUNT(char_name.id) FROM char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN name ON name.id = cast_info.person_id JOIN aka_name ON aka_name.person_id = name.id WHERE name.id > 87660 AND cast_info.person_role_id < 400936 GROUP BY cast_info.id, aka_name.name_pcode_nf, char_name.name, name.name_pcode_nf ORDER BY char_name.name ASC, cast_info.id ASC, name.name_pcode_nf ASC;
