SELECT keyword.keyword, movie_keyword.movie_id FROM movie_keyword JOIN keyword ON keyword.id = movie_keyword.keyword_id WHERE movie_keyword.id >= 482250 AND movie_keyword.movie_id <> 2386990 AND movie_keyword.keyword_id < 266 AND keyword.phonetic_code = 'N6253' ORDER BY movie_keyword.movie_id ASC, keyword.keyword DESC;
