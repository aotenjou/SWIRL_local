select movie_link.link_type_id, movie_info.info_type_id, movie_info.movie_id, min(title.episode_nr) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_link.link_type_id = 4 AND title.id > 724807 AND title.series_years >= '2009-????' group by movie_info.info_type_id, movie_link.link_type_id, movie_info.movie_id having min(title.episode_nr) >= 237;
