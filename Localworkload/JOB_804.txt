select name.name, title.series_years, aka_title.kind_id, aka_title.movie_id, name.name_pcode_cf from kind_type JOIN aka_title ON aka_title.kind_id = kind_type.id JOIN title ON title.id = aka_title.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN name ON name.id = cast_info.person_id where title.phonetic_code < 'J5' AND aka_title.season_nr = 38 order by name.name_pcode_cf DESC, name.name DESC;
