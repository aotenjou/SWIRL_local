select cast_info.id, char_name.md5sum, role_type.id from role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id where cast_info.note > '(credit only) (as J)' AND char_name.md5sum <= 'd6e448de4c2427dd0c0b2b06ffa4d351' AND role_type.role = 'cinematographer' order by char_name.md5sum DESC, role_type.id DESC;
