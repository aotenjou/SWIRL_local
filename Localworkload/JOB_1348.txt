select title.season_nr, title.episode_of_id, link_type.id, movie_link.id, max(movie_companies.note) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_companies.note > '(2013) (Hong Kong) (DVD) (Blu-ray)' AND movie_link.link_type_id = 4 group by title.season_nr, title.episode_of_id, link_type.id, movie_link.id order by link_type.id DESC, title.season_nr DESC, movie_link.id ASC;
