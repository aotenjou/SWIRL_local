select movie_info.info, aka_title.id, title.season_nr from movie_info JOIN title ON title.id = movie_info.movie_id JOIN aka_title ON aka_title.movie_id = title.id where aka_title.kind_id > 3 AND movie_info.movie_id <= 132249 AND title.md5sum < 'e48446153e7e52607f8b942f22f446db' AND aka_title.title <= 'Minäkö isä!' order by title.season_nr DESC, aka_title.id DESC;
