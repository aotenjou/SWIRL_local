select title.season_nr, movie_link.link_type_id, link_type.id, title.kind_id, max(movie_companies.id) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_companies.company_id > 92 AND movie_companies.note = '(1914) (USA) (theatrical)' group by title.season_nr, movie_link.link_type_id, link_type.id, title.kind_id order by movie_link.link_type_id DESC, title.season_nr DESC, title.kind_id ASC;
