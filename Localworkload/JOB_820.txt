SELECT complete_cast.subject_id, aka_title.kind_id, title.md5sum, kind_type.id FROM complete_cast JOIN title ON title.id = complete_cast.movie_id JOIN aka_title ON aka_title.movie_id = title.id JOIN kind_type ON kind_type.id = aka_title.kind_id WHERE aka_title.note < '(International: English title) (imdb display title)' AND aka_title.title <> 'Minäkö isä!' AND complete_cast.status_id <> 4 AND aka_title.id <= 120193 ORDER BY title.md5sum ASC;
