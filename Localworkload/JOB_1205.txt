select char_name.id, cast_info.person_role_id, role_type.id from role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id where char_name.md5sum < 'c71bcc30ec4027737dbc83a044050e94' AND role_type.id >= 9 AND cast_info.note = '(uncredited) (archive footage)' order by char_name.id DESC, role_type.id ASC;
