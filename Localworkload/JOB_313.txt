select cast_info.note, title.title, cast_info.nr_order, complete_cast.subject_id, sum(cast_info.role_id) from char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where char_name.surname_pcode = 'O42' AND complete_cast.subject_id > 2 group by cast_info.note, title.title, complete_cast.subject_id, cast_info.nr_order having sum(cast_info.role_id) = 2427152;
