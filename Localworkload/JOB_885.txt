select kind_type.id, aka_title.imdb_index, title.series_years, name.name_pcode_nf, cast_info.role_id from kind_type JOIN aka_title ON aka_title.kind_id = kind_type.id JOIN title ON title.id = aka_title.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN name ON name.id = cast_info.person_id where aka_title.episode_of_id < 14559 AND aka_title.season_nr = 38 order by aka_title.imdb_index DESC, title.series_years DESC;
