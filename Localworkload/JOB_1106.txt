select title.episode_of_id, movie_info.info_type_id, title.title, max(title.episode_nr) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where title.title < 'Een jaar later' AND title.id > 1451880 AND title.series_years >= '2009-????' group by title.episode_of_id, movie_info.info_type_id, title.title having max(title.episode_nr) >= 237;
