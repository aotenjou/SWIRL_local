SELECT complete_cast.movie_id, title.episode_nr, movie_keyword.keyword_id, comp_cast_type.id FROM movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id WHERE comp_cast_type.id >= 2 AND comp_cast_type.kind > 'crew' AND title.episode_nr <= 29;
