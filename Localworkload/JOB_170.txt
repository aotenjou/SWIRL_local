select movie_keyword.movie_id, title.imdb_index, movie_link.id, movie_keyword.keyword_id from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_link.linked_movie_id < 1808225 AND link_type.link >= 'spin off from' AND title.title >= '(#1.5035)' order by movie_link.id ASC, movie_keyword.movie_id DESC, movie_keyword.keyword_id DESC, title.imdb_index ASC;
