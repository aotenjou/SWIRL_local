select title.episode_of_id, movie_link.link_type_id, title.md5sum, max(title.episode_nr) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.info_type_id < 105 AND title.id > 724807 AND movie_info.movie_id >= 2509321 group by title.episode_of_id, movie_link.link_type_id, title.md5sum having max(title.episode_nr) >= 237;
