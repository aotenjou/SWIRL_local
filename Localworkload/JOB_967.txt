select kind_type.id, movie_companies.note, aka_title.imdb_index, title.season_nr from movie_companies JOIN title ON title.id = movie_companies.movie_id JOIN kind_type ON kind_type.id = title.kind_id JOIN aka_title ON aka_title.kind_id = kind_type.id where kind_type.kind <= 'episode' AND aka_title.imdb_index = 'XI' AND aka_title.production_year >= 1949;
