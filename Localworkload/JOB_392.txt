SELECT title.production_year, movie_link.linked_movie_id, movie_companies.company_type_id, link_type.link FROM link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id WHERE movie_companies.note <> '(1979) (USA: Midwest) (theatrical) (dubbed)' AND title.episode_of_id >= 806041 AND movie_companies.company_id > 708 ORDER BY link_type.link ASC, movie_link.linked_movie_id ASC, title.production_year ASC, movie_companies.company_type_id ASC;
