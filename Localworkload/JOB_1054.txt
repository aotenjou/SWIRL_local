select movie_link.id, link_type.link, movie_keyword.id, title.id from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_link.linked_movie_id = 1808225 AND title.title > '(2000-05-02)' AND title.title >= '(#1.5035)' order by movie_keyword.id ASC, link_type.link DESC, title.id DESC, movie_link.id ASC;
