select char_name.id, title.production_year, title.md5sum, complete_cast.subject_id, sum(complete_cast.movie_id) from char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where cast_info.note = '(set location manager)' AND complete_cast.subject_id > 2 group by title.md5sum, title.production_year, char_name.id, complete_cast.subject_id having sum(complete_cast.movie_id) = 2427152;
