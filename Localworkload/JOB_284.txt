SELECT movie_info_idx.id, name.id, cast_info.id, title.title, aka_name.id, MIN(movie_info_idx.id) FROM movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN name ON name.id = cast_info.person_id JOIN aka_name ON aka_name.person_id = name.id WHERE cast_info.movie_id <> 53429 AND movie_info_idx.info = '179' GROUP BY movie_info_idx.id, name.id, cast_info.id, title.title, aka_name.id;
