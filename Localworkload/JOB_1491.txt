SELECT keyword.phonetic_code, movie_keyword.movie_id, SUM(movie_keyword.movie_id), MAX(movie_keyword.keyword_id) FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id WHERE movie_keyword.id <> 1941574 AND keyword.id <> 43184 AND keyword.phonetic_code <= 'N6253' GROUP BY keyword.phonetic_code, movie_keyword.movie_id ORDER BY keyword.phonetic_code DESC, movie_keyword.movie_id ASC;
