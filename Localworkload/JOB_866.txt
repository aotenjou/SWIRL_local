select link_type.link, title.season_nr, movie_link.movie_id, avg(movie_link.id), max(title.imdb_index) from link_type, movie_link, title where title.production_year <= 1977 AND movie_link.movie_id > 8087 AND movie_link.id = 15293 group by link_type.link, title.season_nr, movie_link.movie_id order by movie_link.movie_id DESC;
