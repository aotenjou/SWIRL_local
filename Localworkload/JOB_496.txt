select title.season_nr, title.kind_id, movie_companies.company_id, movie_companies.note, max(title.episode_of_id) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_link.id <= 16299 AND movie_companies.note = '(1914) (USA) (theatrical)' group by movie_companies.company_id, title.season_nr, movie_companies.note, title.kind_id order by movie_companies.note DESC, title.season_nr DESC, title.kind_id ASC;
