select title.imdb_index, title.phonetic_code, kind_type.id from movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN kind_type ON kind_type.id = title.kind_id where title.season_nr <= 13 AND movie_info_idx.id < 412529 AND title.kind_id <= 2 AND movie_info_idx.info = '1.0.122.02' order by title.phonetic_code DESC, kind_type.id ASC;
