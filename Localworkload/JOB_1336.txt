SELECT person_info.info, movie_info.id, title.series_years, info_type.info FROM title JOIN movie_info ON movie_info.movie_id = title.id JOIN info_type ON info_type.id = movie_info.info_type_id JOIN person_info ON person_info.info_type_id = info_type.id WHERE title.production_year > 1947 AND info_type.info <= 'agent address' AND title.kind_id < 1 AND person_info.info_type_id < 24 ORDER BY title.series_years DESC, person_info.info DESC;
