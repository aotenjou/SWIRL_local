SELECT cast_info.id, info_type.info, name.id, person_info.person_id, SUM(cast_info.nr_order), SUM(info_type.id), COUNT(person_info.info_type_id) FROM cast_info JOIN name ON name.id = cast_info.person_id JOIN person_info ON person_info.person_id = name.id JOIN info_type ON info_type.id = person_info.info_type_id WHERE name.name < 'Boehm, Isabella' GROUP BY cast_info.id, info_type.info, name.id, person_info.person_id HAVING SUM(cast_info.nr_order) <= 9 ORDER BY SUM(cast_info.nr_order) DESC;
