SELECT company_name.md5sum, title.id, movie_companies.company_type_id, movie_info.info, info_type.id FROM info_type JOIN movie_info ON movie_info.info_type_id = info_type.id JOIN title ON title.id = movie_info.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_name ON company_name.id = movie_companies.company_id WHERE movie_companies.id < 1721230 AND title.imdb_index < 'II' AND movie_info.movie_id > 1807954;
