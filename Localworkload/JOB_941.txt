select kind_type.kind, title.episode_nr, movie_info_idx.info_type_id from movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN kind_type ON kind_type.id = title.kind_id where title.episode_nr <= 237 AND movie_info_idx.id < 412529 AND title.kind_id <= 2 AND title.phonetic_code = 'D42' order by title.episode_nr DESC, movie_info_idx.info_type_id ASC;
