SELECT title.phonetic_code, keyword.keyword, movie_keyword.keyword_id, kind_type.id FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN kind_type ON kind_type.id = title.kind_id WHERE title.series_years < '2009-????' AND title.kind_id <> 1 AND keyword.id >= 22473 ORDER BY title.phonetic_code DESC, keyword.keyword ASC, kind_type.id DESC, movie_keyword.keyword_id DESC;
