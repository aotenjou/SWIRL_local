select link_type.link, title.season_nr, movie_link.movie_id, avg(movie_link.id), max(title.imdb_index) from link_type, movie_link, title where movie_link.movie_id <= 138672 AND movie_link.id > 18214 AND title.production_year = 1947 group by link_type.link, title.season_nr, movie_link.movie_id order by movie_link.movie_id DESC;
