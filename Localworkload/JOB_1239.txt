select person_info.info, max(person_info.note), min(person_info.person_id) from person_info where person_info.info > 'Music Album "Admitted" in March of 2011.' AND person_info.person_id < 1290899 AND person_info.info_type_id >= 24 group by person_info.info having min(person_info.person_id) = 1823584 AND max(person_info.note) < 'Anonymous' order by min(person_info.person_id) ASC, max(person_info.note) DESC;
