SELECT char_name.imdb_index, title.title, cast_info.role_id, kind_type.kind, COUNT(cast_info.person_id) FROM char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN kind_type ON kind_type.id = title.kind_id WHERE char_name.surname_pcode = 'F653' AND kind_type.id <> 4 GROUP BY char_name.imdb_index, title.title, cast_info.role_id, kind_type.kind ORDER BY char_name.imdb_index DESC, kind_type.kind ASC, title.title ASC;
