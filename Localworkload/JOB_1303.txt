SELECT aka_title.season_nr, title.title, movie_companies.company_type_id, MIN(movie_companies.note), MAX(title.production_year) FROM movie_companies JOIN title ON title.id = movie_companies.movie_id JOIN aka_title ON aka_title.movie_id = title.id WHERE title.kind_id = 2 AND title.id <= 1931226 GROUP BY aka_title.season_nr, title.title, movie_companies.company_type_id HAVING MAX(title.production_year) = 1986 AND MIN(movie_companies.note) >= '(2012) (USA) (video)' ORDER BY MIN(movie_companies.note) DESC;
