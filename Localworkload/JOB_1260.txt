select title.series_years, title.episode_of_id, movie_info.id, movie_info.info, sum(title.episode_nr) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.info_type_id >= 107 AND movie_info.movie_id <= 2342737 AND title.md5sum < '6feeed89f6a89b5e9c266b5394c332be' group by movie_info.id, title.series_years, movie_info.info, title.episode_of_id;
