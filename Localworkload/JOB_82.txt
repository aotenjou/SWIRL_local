SELECT person_info.info_type_id, name.name_pcode_nf, COUNT(person_info.info_type_id) FROM person_info JOIN name ON name.id = person_info.person_id WHERE person_info.info_type_id <= 22 AND name.id <> 277502 AND name.surname_pcode >= 'M626' AND person_info.info < '(2004) Movie Trailer: "Crash" Digital Imaging: iO FILM' AND name.name >= 'Rouse, Steve' GROUP BY person_info.info_type_id, name.name_pcode_nf;
