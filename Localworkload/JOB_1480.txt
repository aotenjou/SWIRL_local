SELECT movie_companies.company_type_id, company_type.kind FROM movie_companies JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE movie_companies.company_id >= 3351 AND company_type.kind < 'special effects companies' AND movie_companies.company_type_id <= 1 AND company_type.id < 4 AND movie_companies.note = '(????) (USA) (VHS)' ORDER BY company_type.kind ASC, movie_companies.company_type_id DESC;
