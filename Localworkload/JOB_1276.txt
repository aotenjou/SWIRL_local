select movie_info.info, movie_info.id, movie_info.note from movie_info JOIN title ON title.id = movie_info.movie_id JOIN aka_title ON aka_title.movie_id = title.id where aka_title.kind_id = 7 AND movie_info.movie_id < 1792342 AND aka_title.episode_of_id < 14559 AND aka_title.title <= 'Minäkö isä!' order by movie_info.note DESC, movie_info.id DESC;
