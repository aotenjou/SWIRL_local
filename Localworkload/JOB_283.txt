SELECT title.production_year, company_type.id, complete_cast.status_id, movie_companies.movie_id, AVG(title.episode_of_id) FROM complete_cast JOIN title ON title.id = complete_cast.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE movie_companies.movie_id <> 1114010 AND title.phonetic_code > 'Z2521' AND title.kind_id < 2 GROUP BY title.production_year, company_type.id, complete_cast.status_id, movie_companies.movie_id;
