SELECT title.id, role_type.role, cast_info.nr_order, movie_link.linked_movie_id, AVG(cast_info.person_role_id) FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN title ON title.id = cast_info.movie_id JOIN movie_link ON movie_link.movie_id = title.id WHERE title.season_nr <= 13 AND title.phonetic_code <> 'E5132' AND title.md5sum > 'd0433eff32fba157a56109080385abcf' GROUP BY title.id, role_type.role, cast_info.nr_order, movie_link.linked_movie_id;
