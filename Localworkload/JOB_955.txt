SELECT aka_name.person_id, cast_info.movie_id, name.md5sum, role_type.role FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN name ON name.id = cast_info.person_id JOIN aka_name ON aka_name.person_id = name.id WHERE cast_info.movie_id > 2123163 AND role_type.role = 'actress' AND name.name <= 'Ruiz, Diana Jimena' ORDER BY aka_name.person_id ASC, role_type.role ASC;
