SELECT movie_info_idx.movie_id, person_info.info, info_type.info, MAX(person_info.note), AVG(info_type.id) FROM movie_info_idx JOIN info_type ON info_type.id = movie_info_idx.info_type_id JOIN person_info ON person_info.info_type_id = info_type.id WHERE person_info.person_id = 1889740 AND movie_info_idx.movie_id <> 1313571 GROUP BY movie_info_idx.movie_id, person_info.info, info_type.info HAVING MAX(person_info.note) <> 'mcktusk@earlink.net' ORDER BY AVG(info_type.id) ASC;
