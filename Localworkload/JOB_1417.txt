SELECT company_name.name_pcode_nf, movie_companies.note, company_type.id FROM company_type JOIN movie_companies ON movie_companies.company_type_id = company_type.id JOIN company_name ON company_name.id = movie_companies.company_id WHERE movie_companies.id <= 1721230 AND company_name.country_code <> '[in]' AND movie_companies.company_id <= 95180 AND company_type.id >= 3 AND company_name.id >= 159463;
