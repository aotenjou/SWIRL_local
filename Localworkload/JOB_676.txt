SELECT company_type.id, cast_info.role_id, title.phonetic_code, movie_companies.note FROM cast_info JOIN title ON title.id = cast_info.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE movie_companies.movie_id > 2284632 AND title.production_year < 1902 AND title.season_nr <= 3 ORDER BY company_type.id ASC;
