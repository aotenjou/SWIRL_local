select title.title, complete_cast.subject_id, movie_info.movie_id, movie_info_idx.info_type_id, info_type.info from movie_info JOIN info_type ON info_type.id = movie_info.info_type_id JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id JOIN title ON title.id = movie_info_idx.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where info_type.info >= 'LD quality program' AND movie_info_idx.movie_id = 2010299 order by complete_cast.subject_id DESC;
