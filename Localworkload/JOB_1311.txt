SELECT title.title, complete_cast.status_id, comp_cast_type.kind, movie_info.info, SUM(title.kind_id) FROM comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_info ON movie_info.movie_id = title.id WHERE movie_info.info_type_id >= 107 AND movie_info.movie_id <= 2342737 AND title.md5sum < '6feeed89f6a89b5e9c266b5394c332be' GROUP BY title.title, complete_cast.status_id, comp_cast_type.kind, movie_info.info;
