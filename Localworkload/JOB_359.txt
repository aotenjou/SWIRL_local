SELECT cast_info.nr_order, char_name.md5sum, role_type.role FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id WHERE char_name.name >= 'Jacques Clouseau' AND cast_info.id >= 16954885 AND cast_info.role_id > 6 ORDER BY char_name.md5sum ASC, cast_info.nr_order ASC, role_type.role DESC;
