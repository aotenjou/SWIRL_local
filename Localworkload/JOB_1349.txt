SELECT title.production_year, movie_info_idx.id, movie_keyword.movie_id, info_type.id FROM movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_info_idx ON movie_info_idx.movie_id = title.id JOIN info_type ON info_type.id = movie_info_idx.info_type_id WHERE title.kind_id <= 2 AND info_type.id >= 63 AND title.imdb_index >= 'II' AND movie_info_idx.info_type_id = 100;
