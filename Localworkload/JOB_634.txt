SELECT movie_info_idx.info_type_id, title.production_year, person_info.note, info_type.info FROM person_info JOIN info_type ON info_type.id = person_info.info_type_id JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id JOIN title ON title.id = movie_info_idx.movie_id WHERE title.imdb_index < 'II' AND movie_info_idx.id >= 1214182 AND title.episode_of_id <= 838473 AND title.title <= 'Skandalnoye proishestviye v Brikmille' ORDER BY movie_info_idx.info_type_id DESC, title.production_year ASC, person_info.note DESC, info_type.info DESC;
