select title.id, complete_cast.subject_id, keyword.id, comp_cast_type.kind, keyword.keyword from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id where keyword.phonetic_code >= 'N6253' AND complete_cast.movie_id <= 2427152 AND movie_keyword.id >= 2638478 order by complete_cast.subject_id DESC, keyword.id DESC;
