select link_type.link, title.season_nr, movie_link.movie_id, avg(movie_link.id), max(movie_link.movie_id) from link_type, movie_link, title where title.imdb_index <= 'II' AND movie_link.id > 18214 AND title.production_year = 2002 group by link_type.link, title.season_nr, movie_link.movie_id order by movie_link.movie_id DESC;
