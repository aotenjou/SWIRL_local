SELECT movie_info.info_type_id, info_type.info, person_info.person_id FROM movie_info JOIN info_type ON info_type.id = movie_info.info_type_id JOIN person_info ON person_info.info_type_id = info_type.id WHERE movie_info.info = 'MET:600 m' AND movie_info.movie_id = 2096907 AND person_info.info > '"Orson Welles" (1992 -- Los Angeles production; also directed)' AND person_info.person_id <= 1644273 ORDER BY person_info.person_id DESC, info_type.info ASC;
