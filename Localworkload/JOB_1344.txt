SELECT title.md5sum, role_type.role, movie_info_idx.info, cast_info.person_id, MIN(title.md5sum), MIN(title.imdb_index), COUNT(cast_info.nr_order), MAX(movie_info_idx.info_type_id), MAX(cast_info.role_id) FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN title ON title.id = cast_info.movie_id JOIN movie_info_idx ON movie_info_idx.movie_id = title.id WHERE title.series_years <> '2009-????' AND cast_info.nr_order > 5 GROUP BY title.md5sum, role_type.role, movie_info_idx.info, cast_info.person_id;
