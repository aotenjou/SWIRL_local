SELECT movie_keyword.keyword_id, title.episode_of_id, movie_companies.company_id FROM movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id WHERE title.episode_nr <= 54 AND movie_companies.id <> 500250 AND movie_companies.note < '(2005) (USA) (TV) (re-runs)' ORDER BY title.episode_of_id DESC, movie_companies.company_id ASC;
