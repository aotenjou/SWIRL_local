select title.title, complete_cast.subject_id, title.season_nr, movie_info_idx.info_type_id, info_type.info from movie_info JOIN info_type ON info_type.id = movie_info.info_type_id JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id JOIN title ON title.id = movie_info_idx.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where title.md5sum > '05b0479cac0b762d64a8fbab8106e2e3' AND movie_info_idx.movie_id = 2010299 order by complete_cast.subject_id DESC;
