select keyword.phonetic_code, movie_keyword.movie_id, sum(keyword.id), max(movie_keyword.keyword_id) from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id where movie_keyword.movie_id <= 1760735 AND keyword.id = 43184 AND keyword.phonetic_code <= 'N6253' group by keyword.phonetic_code, movie_keyword.movie_id order by keyword.phonetic_code DESC, movie_keyword.movie_id ASC;
