select title.episode_nr, title.phonetic_code, title.imdb_index, count(title.episode_of_id), max(movie_keyword.keyword_id), sum(movie_keyword.movie_id) from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id where title.md5sum < '4cde064e298247ed97cfbe03bb39e3fc' AND keyword.phonetic_code >= 'P4352' AND keyword.keyword = 'year-2078' group by title.episode_nr, title.phonetic_code, title.imdb_index order by title.episode_nr ASC;
