SELECT title.season_nr, cast_info.person_id, name.name_pcode_cf, aka_title.episode_nr, kind_type.kind FROM kind_type JOIN aka_title ON aka_title.kind_id = kind_type.id JOIN title ON title.id = aka_title.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN name ON name.id = cast_info.person_id WHERE title.title > 'Rahma' AND title.episode_of_id <> 384768 AND cast_info.note < '(producer: FUNimation Entertainment)' AND title.episode_nr > 29;
