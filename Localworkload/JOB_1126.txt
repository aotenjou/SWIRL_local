SELECT title.imdb_index, movie_link.link_type_id, keyword.keyword, movie_keyword.id FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id WHERE movie_link.movie_id = 132440 AND title.production_year >= 1903 AND movie_link.id < 16299 ORDER BY movie_keyword.id ASC, title.imdb_index ASC;
