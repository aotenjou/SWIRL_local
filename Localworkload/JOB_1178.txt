select title.season_nr, movie_companies.company_type_id, title.phonetic_code, movie_companies.note, min(movie_companies.movie_id) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where link_type.id > 9 AND movie_companies.note = '(1914) (USA) (theatrical)' group by title.phonetic_code, movie_companies.company_type_id, title.season_nr, movie_companies.note order by movie_companies.note DESC, title.season_nr DESC, title.phonetic_code ASC;
