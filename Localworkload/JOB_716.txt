select movie_keyword.movie_id, title.episode_nr, movie_keyword.id, link_type.link from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_keyword.movie_id < 1908764 AND link_type.link >= 'spin off from' AND title.title >= '(#1.5035)' order by movie_keyword.id ASC, movie_keyword.movie_id DESC, link_type.link DESC, title.episode_nr ASC;
