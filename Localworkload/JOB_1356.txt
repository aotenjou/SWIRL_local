select title.phonetic_code, complete_cast.subject_id, title.production_year, movie_keyword.id, title.episode_of_id from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id where complete_cast.subject_id < 2 AND keyword.keyword <= 'civic-pride' AND complete_cast.movie_id < 1438895 order by movie_keyword.id DESC, title.episode_of_id DESC;
