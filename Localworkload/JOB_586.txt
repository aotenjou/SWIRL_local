select movie_link.id, title.episode_of_id, title.series_years, max(title.episode_nr) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.info_type_id < 105 AND title.id > 724807 AND movie_info.movie_id >= 2369921 group by title.episode_of_id, movie_link.id, title.series_years having max(title.episode_nr) >= 237;
