SELECT movie_companies.note, company_name.md5sum, SUM(movie_companies.company_id), MAX(movie_companies.note), COUNT(movie_companies.id), COUNT(movie_companies.company_type_id) FROM company_name, movie_companies WHERE company_name.name_pcode_sf > 'A3463' AND company_name.id = 33018 AND movie_companies.note >= '(2005) (USA) (TV) (re-runs)' GROUP BY movie_companies.note, company_name.md5sum ORDER BY SUM(movie_companies.company_id) ASC;
