select movie_link.link_type_id, title.season_nr, title.md5sum, avg(movie_link.linked_movie_id), min(movie_link.movie_id) from link_type, movie_link, title where title.imdb_index <= 'I' AND movie_link.id > 18214 AND title.production_year = 1947 group by title.md5sum, title.season_nr, movie_link.link_type_id order by title.season_nr DESC;
