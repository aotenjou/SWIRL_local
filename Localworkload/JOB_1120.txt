select person_info.info, min(person_info.note), max(person_info.person_id) from person_info where person_info.info = 'Has a little sister called Heather Quinn.' AND person_info.person_id < 1290899 AND person_info.info_type_id >= 24 group by person_info.info having min(person_info.note) = 'Anonymous' AND max(person_info.person_id) < 2472041 order by min(person_info.note) ASC, max(person_info.person_id) DESC;
