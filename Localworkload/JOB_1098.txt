select name.id, title.md5sum, aka_title.imdb_index, cast_info.nr_order, aka_title.kind_id from kind_type JOIN aka_title ON aka_title.kind_id = kind_type.id JOIN title ON title.id = aka_title.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN name ON name.id = cast_info.person_id where aka_title.episode_of_id < 14559 AND aka_title.imdb_index > 'X' order by aka_title.kind_id DESC, aka_title.imdb_index DESC;
