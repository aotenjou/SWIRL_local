SELECT title.series_years, complete_cast.id, movie_keyword.movie_id, SUM(title.production_year), COUNT(title.season_nr) FROM complete_cast JOIN title ON title.id = complete_cast.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id WHERE movie_keyword.id < 3398212 AND title.imdb_index < 'II' AND title.md5sum < '132405233a630aab41fe394628edbeee' AND title.kind_id <= 1 GROUP BY title.series_years, complete_cast.id, movie_keyword.movie_id ORDER BY COUNT(title.season_nr) DESC, SUM(title.production_year) DESC;
