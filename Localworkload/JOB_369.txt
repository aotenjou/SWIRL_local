SELECT title.production_year, name.id, aka_name.id, movie_keyword.keyword_id, cast_info.movie_id FROM aka_name JOIN name ON name.id = aka_name.person_id JOIN cast_info ON cast_info.person_id = name.id JOIN title ON title.id = cast_info.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id WHERE title.id <= 724807 AND movie_keyword.id <> 3992406 ORDER BY title.production_year ASC, aka_name.id DESC;
