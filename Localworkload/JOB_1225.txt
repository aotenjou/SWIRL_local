select kind_type.id, movie_info_idx.info, title.imdb_index from movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN kind_type ON kind_type.id = title.kind_id where movie_info_idx.id <= 1214182 AND title.phonetic_code < 'M212' AND title.kind_id <= 2 AND title.season_nr = 3 order by kind_type.id DESC, movie_info_idx.info ASC;
