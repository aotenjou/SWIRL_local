SELECT name.surname_pcode, aka_name.name, MAX(name.name_pcode_cf), MIN(name.name) FROM name JOIN aka_name ON aka_name.person_id = name.id WHERE name.surname_pcode <> 'B42' AND name.imdb_index < 'III' AND aka_name.id <= 266577 GROUP BY name.surname_pcode, aka_name.name HAVING MIN(name.name) = 'Papazian, Marty' AND MAX(name.name_pcode_cf) > 'A256' ORDER BY aka_name.name DESC, name.surname_pcode ASC;
