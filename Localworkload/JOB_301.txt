SELECT cast_info.person_role_id, title.imdb_index, role_type.role FROM title JOIN cast_info ON cast_info.movie_id = title.id JOIN role_type ON role_type.id = cast_info.role_id WHERE cast_info.movie_id <> 1729887 AND role_type.role = 'actress' AND cast_info.person_role_id = 1197458 ORDER BY title.imdb_index DESC, cast_info.person_role_id DESC, role_type.role ASC;
