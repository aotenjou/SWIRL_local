select char_name.id, title.production_year, title.md5sum, title.episode_nr, sum(complete_cast.status_id) from char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where title.md5sum <= '05b0479cac0b762d64a8fbab8106e2e3' AND complete_cast.subject_id > 2 group by title.md5sum, char_name.id, title.episode_nr, title.production_year having sum(complete_cast.status_id) = 2427152;
