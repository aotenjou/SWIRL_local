SELECT cast_info.nr_order, title.production_year, aka_title.id, kind_type.kind FROM aka_title JOIN kind_type ON kind_type.id = aka_title.kind_id JOIN title ON title.kind_id = kind_type.id JOIN cast_info ON cast_info.movie_id = title.id WHERE aka_title.md5sum < '287a542d59292bc94ae054f698091218' AND cast_info.id <= 26114093 AND kind_type.kind <= 'movie' AND aka_title.phonetic_code = 'M6213' ORDER BY aka_title.id DESC, cast_info.nr_order ASC;
