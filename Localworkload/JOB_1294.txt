select title.title, complete_cast.status_id, comp_cast_type.kind, movie_info.info, sum(title.kind_id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.info_type_id >= 50 AND movie_info.movie_id <= 41076 AND title.md5sum < '923f54faa9d41a1dac963b1e4ba1b44b' group by title.title, complete_cast.status_id, comp_cast_type.kind, movie_info.info;
