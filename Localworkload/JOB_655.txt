select movie_info.movie_id, movie_info.info, complete_cast.status_id, title.md5sum, sum(title.kind_id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.info_type_id >= 65 AND comp_cast_type.kind <= 'complete' AND title.title < '(2002-03-21)' group by movie_info.info, complete_cast.status_id, movie_info.movie_id, title.md5sum;
