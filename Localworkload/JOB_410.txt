SELECT title.phonetic_code, cast_info.role_id, complete_cast.subject_id, role_type.role, comp_cast_type.kind FROM comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN role_type ON role_type.id = cast_info.role_id WHERE complete_cast.status_id >= 4 AND comp_cast_type.id < 1 ORDER BY cast_info.role_id DESC;
