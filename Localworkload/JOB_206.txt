SELECT title.phonetic_code, keyword.keyword, company_name.md5sum, movie_companies.company_type_id, movie_keyword.keyword_id FROM company_name JOIN movie_companies ON movie_companies.company_id = company_name.id JOIN title ON title.id = movie_companies.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id JOIN keyword ON keyword.id = movie_keyword.keyword_id WHERE title.phonetic_code >= 'D42' AND movie_keyword.keyword_id >= 7449 AND title.kind_id <= 1;
