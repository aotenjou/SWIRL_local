SELECT name.gender, info_type.id, aka_name.surname_pcode, person_info.id, MIN(name.id) FROM info_type JOIN person_info ON person_info.info_type_id = info_type.id JOIN name ON name.id = person_info.person_id JOIN aka_name ON aka_name.person_id = name.id WHERE name.id > 2295724 AND name.imdb_index <= 'III' GROUP BY name.gender, info_type.id, aka_name.surname_pcode, person_info.id ORDER BY name.gender ASC;
