select movie_keyword.keyword_id, movie_keyword.id, keyword.keyword, count(movie_keyword.movie_id), max(keyword.phonetic_code), sum(movie_keyword.id) from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id where keyword.keyword < 'fiber-optic-scope' AND title.episode_nr >= 48 AND title.md5sum = 'fb10c3a27024448f6c7b21005d371c1c' group by keyword.keyword, movie_keyword.id, movie_keyword.keyword_id order by keyword.keyword ASC;
