select complete_cast.movie_id, complete_cast.subject_id, keyword.id, comp_cast_type.kind, movie_keyword.id from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id where keyword.keyword >= 'idol-singer' AND title.id <= 123246 AND keyword.phonetic_code >= 'N6253' order by movie_keyword.id DESC, complete_cast.movie_id DESC;
