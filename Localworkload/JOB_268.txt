select title.title, movie_link.linked_movie_id, movie_link.movie_id, title.md5sum from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where movie_keyword.movie_id < 26393 AND link_type.link >= 'spin off from' AND link_type.link >= 'spin off from' order by movie_link.movie_id ASC, movie_link.linked_movie_id DESC, title.md5sum DESC, title.title ASC;
