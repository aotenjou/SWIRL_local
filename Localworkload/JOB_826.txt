select movie_info.movie_id, title.md5sum, link_type.id, movie_link.id from movie_info JOIN title ON title.id = movie_info.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id where title.production_year = 2002 AND title.id >= 75181 AND title.imdb_index < 'I' order by movie_info.movie_id ASC;
