SELECT cast_info.movie_id, aka_name.name_pcode_nf, name.name_pcode_cf, role_type.id FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN name ON name.id = cast_info.person_id JOIN aka_name ON aka_name.person_id = name.id WHERE cast_info.nr_order < 27 AND cast_info.note >= '(managing director: Temple Street Productions)' AND name.name_pcode_nf <> 'M6216' AND aka_name.name_pcode_cf >= 'B6526';
