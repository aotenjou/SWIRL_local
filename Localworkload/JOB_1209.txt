select aka_title.title, aka_title.kind_id, title.season_nr from movie_info JOIN title ON title.id = movie_info.movie_id JOIN aka_title ON aka_title.movie_id = title.id where movie_info.movie_id < 768608 AND movie_info.info = 'Denmark:20 March 1935' AND title.md5sum < 'e48446153e7e52607f8b942f22f446db' AND aka_title.id <= 361224 order by title.season_nr DESC, aka_title.kind_id DESC;
