SELECT movie_link.linked_movie_id, title.season_nr, movie_companies.company_type_id, link_type.link, SUM(link_type.id), SUM(movie_link.id) FROM movie_companies JOIN title ON title.id = movie_companies.movie_id JOIN movie_link ON movie_link.movie_id = title.id JOIN link_type ON link_type.id = movie_link.link_type_id WHERE movie_link.movie_id > 140768 AND title.id <> 565537 GROUP BY movie_link.linked_movie_id, title.season_nr, movie_companies.company_type_id, link_type.link HAVING SUM(link_type.id) > 9;
