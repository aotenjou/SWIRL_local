SELECT title.production_year, movie_keyword.movie_id, movie_companies.company_type_id, keyword.id, COUNT(keyword.id) FROM movie_companies JOIN title ON title.id = movie_companies.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id JOIN keyword ON keyword.id = movie_keyword.keyword_id WHERE movie_companies.movie_id <= 1951183 AND title.phonetic_code <= 'S5215' GROUP BY title.production_year, movie_keyword.movie_id, movie_companies.company_type_id, keyword.id ORDER BY COUNT(keyword.id) DESC;
