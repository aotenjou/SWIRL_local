select title.episode_of_id, title.series_years, movie_info.movie_id, max(movie_link.id) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where title.episode_nr < 237 AND title.id > 724807 AND movie_info.info_type_id >= 105 group by title.episode_of_id, movie_info.movie_id, title.series_years having max(movie_link.id) >= 23398;
