SELECT complete_cast.id, title.season_nr, keyword.id, movie_keyword.movie_id, comp_cast_type.kind FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id WHERE title.title <> 'A Cover of Art' AND complete_cast.movie_id >= 1802474 AND keyword.id <= 86959;
