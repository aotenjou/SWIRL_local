SELECT title.id, movie_info_idx.info_type_id, comp_cast_type.kind, complete_cast.id, SUM(complete_cast.subject_id) FROM movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id WHERE movie_info_idx.info < '2..11111.3' AND title.season_nr > 3 GROUP BY title.id, movie_info_idx.info_type_id, comp_cast_type.kind, complete_cast.id HAVING SUM(complete_cast.subject_id) > 1;
