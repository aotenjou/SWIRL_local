SELECT title.season_nr, company_name.name_pcode_nf, movie_info_idx.id, movie_companies.company_id, AVG(movie_companies.movie_id) FROM movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_name ON company_name.id = movie_companies.company_id WHERE movie_info_idx.id > 526904 AND movie_companies.id < 1765864 GROUP BY title.season_nr, company_name.name_pcode_nf, movie_info_idx.id, movie_companies.company_id HAVING AVG(movie_companies.movie_id) > 1951183 ORDER BY company_name.name_pcode_nf DESC, title.season_nr DESC;
