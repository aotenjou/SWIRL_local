select movie_keyword.movie_id, keyword.phonetic_code from keyword, movie_keyword where movie_keyword.keyword_id >= 2987 AND keyword.id = 22473 AND movie_keyword.movie_id >= 2082380 AND movie_keyword.id > 735536 AND keyword.phonetic_code < 'C4265' order by movie_keyword.movie_id DESC, keyword.phonetic_code DESC;
