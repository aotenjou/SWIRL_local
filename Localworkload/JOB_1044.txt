SELECT cast_info.person_id, role_type.id, MIN(role_type.role), MAX(cast_info.note) FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id WHERE cast_info.role_id < 11 AND cast_info.id <= 24473479 AND cast_info.person_role_id <= 2373 AND role_type.id > 7 AND cast_info.nr_order <= 27 GROUP BY cast_info.person_id, role_type.id ORDER BY role_type.id DESC, cast_info.person_id ASC;
