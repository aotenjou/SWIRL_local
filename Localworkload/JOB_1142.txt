SELECT title.production_year, role_type.role, cast_info.id, movie_keyword.movie_id, keyword.keyword, AVG(keyword.id), SUM(movie_keyword.keyword_id) FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN role_type ON role_type.id = cast_info.role_id WHERE role_type.id <> 9 AND cast_info.nr_order >= 69 GROUP BY title.production_year, role_type.role, cast_info.id, movie_keyword.movie_id, keyword.keyword;
