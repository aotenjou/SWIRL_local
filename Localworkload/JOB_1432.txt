select char_name.id, cast_info.person_role_id, role_type.id from role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id where char_name.md5sum < '6c6fc3f5ca5ddd7cd67e799565ee1d46' AND role_type.id >= 7 AND cast_info.person_id = 643234 order by char_name.id DESC, role_type.id ASC;
