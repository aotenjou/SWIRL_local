SELECT person_info.id, name.name, cast_info.person_role_id, movie_info_idx.info_type_id, info_type.id FROM cast_info JOIN name ON name.id = cast_info.person_id JOIN person_info ON person_info.person_id = name.id JOIN info_type ON info_type.id = person_info.info_type_id JOIN movie_info_idx ON movie_info_idx.info_type_id = info_type.id WHERE person_info.note < 'MilleniumBabyGirl' AND cast_info.nr_order >= 27 ORDER BY info_type.id DESC, cast_info.person_role_id ASC, name.name DESC, movie_info_idx.info_type_id ASC;
