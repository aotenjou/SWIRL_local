SELECT title.season_nr, movie_keyword.id, keyword.phonetic_code, complete_cast.status_id FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id WHERE title.id < 96842 AND complete_cast.movie_id < 1923775 AND title.imdb_index >= 'II' AND keyword.phonetic_code > 'N6253' ORDER BY title.season_nr ASC, complete_cast.status_id DESC;
