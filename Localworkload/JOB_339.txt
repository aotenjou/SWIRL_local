SELECT movie_link.movie_id, link_type.id, title.series_years, cast_info.person_id, role_type.id FROM link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN role_type ON role_type.id = cast_info.role_id WHERE role_type.id = 9 AND title.season_nr > 3 AND role_type.role >= 'cinematographer' AND movie_link.link_type_id = 15;
