select char_name.id, title.title, cast_info.note, complete_cast.subject_id, sum(complete_cast.movie_id) from char_name JOIN cast_info ON cast_info.person_role_id = char_name.id JOIN title ON title.id = cast_info.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id where cast_info.note = '(publicist)' AND complete_cast.subject_id > 2 group by char_name.id, title.title, complete_cast.subject_id, cast_info.note having sum(complete_cast.movie_id) = 2427152;
