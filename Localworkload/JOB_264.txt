select title.season_nr, movie_link.link_type_id, link_type.id, title.episode_of_id, max(movie_companies.note) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_link.id > 16299 AND movie_companies.note = '(1914) (USA) (theatrical)' group by movie_link.link_type_id, title.season_nr, link_type.id, title.episode_of_id order by title.episode_of_id DESC, title.season_nr DESC, link_type.id ASC;
