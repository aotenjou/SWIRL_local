SELECT title.kind_id, movie_companies.company_type_id, movie_info.movie_id, info_type.id FROM movie_companies JOIN title ON title.id = movie_companies.movie_id JOIN movie_info ON movie_info.movie_id = title.id JOIN info_type ON info_type.id = movie_info.info_type_id WHERE movie_info.note < '(Chicago International Film Festival)' AND info_type.id <> 63 AND title.phonetic_code > 'U5263' ORDER BY movie_info.movie_id ASC;
