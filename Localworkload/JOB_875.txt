select person_info.info, min(person_info.person_id), max(person_info.info_type_id) from person_info where person_info.info = '1952' AND person_info.person_id < 1290899 AND person_info.note >= 'Anonymous' group by person_info.info having min(person_info.person_id) = 1889740 AND max(person_info.info_type_id) < 24 order by min(person_info.person_id) ASC, max(person_info.info_type_id) DESC;
