SELECT movie_keyword.id, keyword.phonetic_code, MIN(movie_keyword.keyword_id), MIN(keyword.phonetic_code) FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id WHERE keyword.id <> 104859 AND movie_keyword.movie_id <= 2065649 AND movie_keyword.id >= 3949487 GROUP BY movie_keyword.id, keyword.phonetic_code ORDER BY movie_keyword.id ASC, keyword.phonetic_code ASC;
