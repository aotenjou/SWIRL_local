select title.episode_of_id, movie_link.id, title.season_nr, movie_companies.note, max(movie_link.link_type_id) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_companies.note > '(producer)' AND link_type.id = 9 group by title.season_nr, title.episode_of_id, movie_companies.note, movie_link.id order by movie_companies.note DESC, title.season_nr DESC, title.episode_of_id ASC;
