SELECT person_info.info_type_id, name.id, info_type.info FROM name JOIN person_info ON person_info.person_id = name.id JOIN info_type ON info_type.id = person_info.info_type_id WHERE name.name > 'Marlow, Rachel' AND info_type.info < 'copyright holder' AND person_info.info <> 'Attended Thurrock Technical College, graduating in 1995 with a City & Guilds Diploma in Motor Vehicle Studies.' AND info_type.id < 8 AND name.id <= 202842 ORDER BY person_info.info_type_id ASC, info_type.info ASC, name.id ASC;
