SELECT name.name, cast_info.person_id, char_name.name FROM name JOIN cast_info ON cast_info.person_id = name.id JOIN char_name ON char_name.id = cast_info.person_role_id WHERE name.gender >= 'm' AND char_name.name_pcode_nf >= 'W6242' AND char_name.md5sum >= 'f085e192eaf186c0e7ba3d9a6edc505c' AND cast_info.movie_id > 1925567 ORDER BY name.name ASC, cast_info.person_id ASC, char_name.name ASC;
