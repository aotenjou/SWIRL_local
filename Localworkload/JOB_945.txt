SELECT title.production_year, movie_info_idx.movie_id, movie_keyword.movie_id FROM movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN movie_info_idx ON movie_info_idx.movie_id = title.id WHERE title.episode_nr > 29 AND title.title <= 'Not Tonight Tweety Pie' AND title.production_year > 1902 ORDER BY movie_keyword.movie_id ASC;
