select title.id, complete_cast.subject_id, title.md5sum, title.episode_of_id, title.series_years from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id where keyword.id >= 22473 AND keyword.keyword <= 'year-2078' AND title.md5sum >= '05b0479cac0b762d64a8fbab8106e2e3' order by title.series_years DESC, title.id DESC;
