select movie_keyword.keyword_id, movie_keyword.id, keyword.keyword, count(movie_keyword.movie_id), max(title.md5sum), sum(movie_keyword.id) from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id where keyword.keyword < 'civic-pride' AND keyword.phonetic_code >= 'N6253' AND title.episode_nr = 29 group by movie_keyword.id, movie_keyword.keyword_id, keyword.keyword order by movie_keyword.keyword_id ASC;
