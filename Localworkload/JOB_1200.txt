select title.episode_nr, keyword.phonetic_code, movie_keyword.id, count(movie_keyword.movie_id), max(keyword.keyword), sum(movie_keyword.id) from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id where keyword.keyword < 'movie-reality' AND movie_keyword.keyword_id >= 85898 AND title.md5sum = '05b0479cac0b762d64a8fbab8106e2e3' group by title.episode_nr, movie_keyword.id, keyword.phonetic_code order by title.episode_nr ASC;
