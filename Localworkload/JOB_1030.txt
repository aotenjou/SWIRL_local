SELECT title.kind_id, movie_companies.movie_id, company_name.name_pcode_nf FROM title JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_name ON company_name.id = movie_companies.company_id WHERE title.title <> 'Flower''s Squirt Shower 4' AND company_name.name < 'WYES' AND movie_companies.company_id <> 708 ORDER BY movie_companies.movie_id DESC, company_name.name_pcode_nf DESC;
