select title.season_nr, movie_link.link_type_id, link_type.id, movie_companies.note, max(title.episode_of_id) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_link.id > 18214 AND movie_companies.note = '(producer)' group by title.season_nr, movie_link.link_type_id, link_type.id, movie_companies.note order by movie_companies.note DESC, title.season_nr DESC, movie_link.link_type_id ASC;
