select title.episode_nr, keyword.id, title.episode_of_id, count(movie_keyword.keyword_id), max(title.md5sum), sum(title.season_nr) from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id where title.series_years <= '2005-2009' AND keyword.phonetic_code >= 'P4352' AND keyword.keyword = 'year-2078' group by title.episode_nr, title.episode_of_id, keyword.id order by title.episode_nr ASC;
