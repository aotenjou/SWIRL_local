select title.episode_of_id, title.episode_nr, title.id, max(movie_info.movie_id) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where title.series_years < '2005-2009' AND movie_info.info_type_id > 50 AND movie_link.id >= 18214 group by title.episode_of_id, title.id, title.episode_nr having max(movie_info.movie_id) >= 695564;
