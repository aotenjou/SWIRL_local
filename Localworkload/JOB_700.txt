SELECT cast_info.nr_order, role_type.id FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id WHERE cast_info.note < '(chief designer: Jann Lee "The Bouncer")' AND cast_info.person_id <> 3945802 AND role_type.role > 'actress' AND cast_info.role_id <> 6 ORDER BY role_type.id ASC, cast_info.nr_order DESC;
