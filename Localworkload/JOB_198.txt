select title.episode_of_id, title.series_years, title.episode_nr, max(movie_info.info_type_id) from movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.movie_id < 2444580 AND title.id > 724807 AND movie_link.id >= 18214 group by title.episode_of_id, title.episode_nr, title.series_years having max(movie_info.info_type_id) >= 65;
