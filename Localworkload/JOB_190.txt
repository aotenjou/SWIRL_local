SELECT title.imdb_index, complete_cast.id, comp_cast_type.id, kind_type.id, MAX(complete_cast.movie_id) FROM kind_type JOIN title ON title.kind_id = kind_type.id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id WHERE title.md5sum > 'ba830df41cf756fd345e2b9683690c19' AND title.phonetic_code <= 'H3162' GROUP BY title.imdb_index, complete_cast.id, comp_cast_type.id, kind_type.id ORDER BY title.imdb_index DESC, comp_cast_type.id ASC, complete_cast.id DESC;
