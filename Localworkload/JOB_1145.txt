SELECT person_info.person_id, name.name_pcode_nf, aka_name.name_pcode_nf FROM person_info JOIN name ON name.id = person_info.person_id JOIN aka_name ON aka_name.person_id = name.id WHERE person_info.id >= 372300 AND name.id <= 202842 AND aka_name.name_pcode_cf > 'M4521' AND aka_name.md5sum <> '7cf23d5fa36a011c05997cbe37cfaf14' ORDER BY name.name_pcode_nf DESC, aka_name.name_pcode_nf ASC;
