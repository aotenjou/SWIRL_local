SELECT title.phonetic_code, movie_companies.movie_id, movie_info_idx.movie_id, company_type.id, COUNT(movie_companies.movie_id) FROM movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE title.episode_nr = 237 AND title.production_year >= 1986 AND movie_companies.movie_id > 880874 GROUP BY title.phonetic_code, movie_companies.movie_id, movie_info_idx.movie_id, company_type.id;
