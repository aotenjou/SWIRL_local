select title.imdb_index, movie_info_idx.info, title.episode_of_id from movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN kind_type ON kind_type.id = title.kind_id where title.season_nr <= 13 AND movie_info_idx.id > 1096919 AND title.kind_id <= 2 AND title.phonetic_code = 'R2161' order by movie_info_idx.info ASC, title.episode_of_id ASC;
