SELECT char_name.id, cast_info.person_role_id, role_type.id FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id WHERE char_name.md5sum < 'c71bcc30ec4027737dbc83a044050e94' AND role_type.id >= 9 AND cast_info.person_id <> 1430034 ORDER BY char_name.id DESC, role_type.id ASC;
