SELECT title.episode_nr, cast_info.note, movie_link.link_type_id, role_type.id, SUM(cast_info.movie_id) FROM movie_link JOIN title ON title.id = movie_link.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN role_type ON role_type.id = cast_info.role_id WHERE movie_link.linked_movie_id <= 2221955 AND title.md5sum = 'd0433eff32fba157a56109080385abcf' GROUP BY title.episode_nr, cast_info.note, movie_link.link_type_id, role_type.id ORDER BY title.episode_nr DESC, role_type.id DESC, cast_info.note DESC, movie_link.link_type_id ASC;
