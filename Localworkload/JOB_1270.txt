select complete_cast.movie_id, title.md5sum, title.episode_nr, complete_cast.status_id, min(movie_keyword.id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id where comp_cast_type.id < 1 AND comp_cast_type.id < 4 group by title.md5sum, complete_cast.status_id, complete_cast.movie_id, title.episode_nr having min(movie_keyword.id) > 2753158 order by title.md5sum DESC;
