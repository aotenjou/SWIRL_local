SELECT company_name.md5sum, title.md5sum, keyword.id, movie_companies.movie_id, movie_keyword.keyword_id FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_name ON company_name.id = movie_companies.company_id WHERE keyword.keyword <= 'year-2078' AND company_name.id = 159463 ORDER BY movie_keyword.keyword_id DESC, company_name.md5sum DESC, keyword.id DESC, title.md5sum ASC;
