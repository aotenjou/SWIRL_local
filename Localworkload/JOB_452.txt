select title.md5sum, complete_cast.movie_id, title.episode_nr, movie_keyword.id, min(complete_cast.status_id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id where comp_cast_type.id < 1 AND comp_cast_type.id < 1 group by complete_cast.movie_id, title.md5sum, movie_keyword.id, title.episode_nr having min(complete_cast.status_id) > 4 order by title.md5sum DESC;
