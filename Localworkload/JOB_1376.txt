select cast_info.person_role_id, name.md5sum, aka_name.name, role_type.id from aka_name JOIN name ON name.id = aka_name.person_id JOIN cast_info ON cast_info.person_id = name.id JOIN role_type ON role_type.id = cast_info.role_id where role_type.role > 'director' AND cast_info.note > '(2006)' AND aka_name.person_id < 2862465 AND role_type.id >= 9;
