select title.imdb_index, movie_link.link_type_id, link_type.id, movie_companies.note, min(title.episode_of_id) from link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id where movie_link.id > 16299 AND title.episode_of_id = 384768 group by movie_companies.note, movie_link.link_type_id, title.imdb_index, link_type.id order by movie_companies.note DESC, link_type.id DESC, movie_link.link_type_id ASC;
