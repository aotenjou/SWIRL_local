select cast_info.person_id, char_name.name, role_type.id from role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id where role_type.role = 'cinematographer' AND role_type.id <= 9 AND cast_info.note = '(runner: director)' order by cast_info.person_id DESC, role_type.id DESC;
