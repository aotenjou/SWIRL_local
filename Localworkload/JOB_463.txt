SELECT title.md5sum, complete_cast.id, kind_type.id, aka_title.title, comp_cast_type.id FROM comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN kind_type ON kind_type.id = title.kind_id JOIN aka_title ON aka_title.kind_id = kind_type.id WHERE title.season_nr <> 13 AND aka_title.kind_id <= 7 AND aka_title.imdb_index < 'X';
