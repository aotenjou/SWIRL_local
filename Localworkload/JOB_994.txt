select movie_info.movie_id, complete_cast.status_id, title.title, movie_info.info_type_id, sum(title.kind_id) from comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_info ON movie_info.movie_id = title.id where movie_info.info >= '$6,886,080 (USA) (9 November 2003)' AND title.md5sum <= 'bbb03311ac842f628147975d1b093de0' AND comp_cast_type.kind < 'complete+verified' group by title.title, complete_cast.status_id, movie_info.info_type_id, movie_info.movie_id;
