SELECT name.name_pcode_cf, cast_info.note, person_info.info_type_id, AVG(cast_info.person_role_id) FROM cast_info JOIN name ON name.id = cast_info.person_id JOIN person_info ON person_info.person_id = name.id WHERE name.name_pcode_cf < 'K52' AND person_info.info >= '"X Magazine" (France), September 1991' AND cast_info.movie_id <= 1355302 GROUP BY name.name_pcode_cf, cast_info.note, person_info.info_type_id ORDER BY AVG(cast_info.person_role_id) DESC;
