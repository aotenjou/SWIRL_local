SELECT link_type.id, title.id, movie_link.id, movie_companies.note, company_type.kind FROM link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE movie_companies.company_type_id <> 1 AND title.kind_id >= 1 ORDER BY movie_link.id DESC, title.id DESC, link_type.id DESC, movie_companies.note ASC;
