SELECT title.season_nr, movie_keyword.movie_id, keyword.phonetic_code, movie_info.note, COUNT(movie_info.info_type_id) FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN movie_info ON movie_info.movie_id = title.id WHERE movie_info.id <> 7849144 AND title.series_years = '1996-????' GROUP BY title.season_nr, movie_keyword.movie_id, keyword.phonetic_code, movie_info.note ORDER BY movie_keyword.movie_id ASC, movie_info.note DESC;
