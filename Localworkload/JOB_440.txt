SELECT title.imdb_index, aka_title.movie_id, comp_cast_type.id, kind_type.id, complete_cast.id FROM kind_type JOIN aka_title ON aka_title.kind_id = kind_type.id JOIN title ON title.id = aka_title.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id WHERE aka_title.kind_id <> 7 AND title.episode_nr >= 237 AND comp_cast_type.id >= 2;
