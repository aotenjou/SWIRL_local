select movie_info.info, aka_title.note, title.production_year from movie_info JOIN title ON title.id = movie_info.movie_id JOIN aka_title ON aka_title.movie_id = title.id where aka_title.episode_of_id < 14559 AND movie_info.info_type_id = 65 AND title.md5sum < 'e48446153e7e52607f8b942f22f446db' AND aka_title.title <= 'Minäkö isä!' order by movie_info.info DESC, aka_title.note DESC;
