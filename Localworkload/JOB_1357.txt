SELECT aka_name.id, name.name_pcode_cf, person_info.person_id, MAX(aka_name.md5sum) FROM aka_name JOIN name ON name.id = aka_name.person_id JOIN person_info ON person_info.person_id = name.id WHERE person_info.info_type_id >= 22 AND aka_name.imdb_index > 'III' AND aka_name.name >= 'Drolette, Gregory Alexander' GROUP BY aka_name.id, name.name_pcode_cf, person_info.person_id;
