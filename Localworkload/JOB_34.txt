SELECT title.episode_of_id, movie_link.movie_id, link_type.id, movie_keyword.id FROM link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id WHERE movie_keyword.id <= 3937290 AND title.season_nr <> 13 AND title.kind_id < 7 ORDER BY title.episode_of_id DESC, movie_keyword.id ASC, link_type.id ASC, movie_link.movie_id ASC;
