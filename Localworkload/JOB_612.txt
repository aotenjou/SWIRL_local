SELECT link_type.link, title.episode_of_id, movie_link.link_type_id, cast_info.person_id, role_type.role FROM link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN cast_info ON cast_info.movie_id = title.id JOIN role_type ON role_type.id = cast_info.role_id WHERE title.kind_id = 1 AND title.production_year <= 1947 ORDER BY title.episode_of_id ASC, cast_info.person_id ASC, link_type.link DESC, role_type.role ASC;
