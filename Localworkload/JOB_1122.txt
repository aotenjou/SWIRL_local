select char_name.imdb_index, cast_info.nr_order, role_type.id from role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id where char_name.name_pcode_nf = 'G6536' AND role_type.id <= 10 AND role_type.role = 'cinematographer' order by char_name.imdb_index DESC, cast_info.nr_order DESC;
