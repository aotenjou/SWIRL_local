SELECT title.md5sum, comp_cast_type.id, complete_cast.movie_id, movie_keyword.id, MIN(complete_cast.status_id) FROM comp_cast_type JOIN complete_cast ON complete_cast.status_id = comp_cast_type.id JOIN title ON title.id = complete_cast.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id WHERE comp_cast_type.id < 1 AND title.episode_nr < 54 GROUP BY title.md5sum, comp_cast_type.id, complete_cast.movie_id, movie_keyword.id HAVING MIN(complete_cast.status_id) > 4 ORDER BY title.md5sum DESC;
