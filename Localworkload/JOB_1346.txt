SELECT name.name_pcode_nf, aka_title.imdb_index, cast_info.id, title.imdb_index, person_info.note FROM person_info JOIN name ON name.id = person_info.person_id JOIN cast_info ON cast_info.person_id = name.id JOIN title ON title.id = cast_info.movie_id JOIN aka_title ON aka_title.movie_id = title.id WHERE title.title > 'Een jaar later' AND title.md5sum >= '132405233a630aab41fe394628edbeee' ORDER BY name.name_pcode_nf ASC, title.imdb_index ASC;
