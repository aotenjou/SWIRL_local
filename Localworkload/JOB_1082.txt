select movie_companies.company_type_id, company_type.kind, count(movie_companies.company_type_id), sum(movie_companies.movie_id), max(company_type.kind) from company_type, movie_companies where movie_companies.movie_id < 1699103 AND movie_companies.id >= 376282 AND movie_companies.note = '(1958) (USA) (TV)' group by movie_companies.company_type_id, company_type.kind;
