SELECT movie_link.movie_id, title.imdb_index, movie_keyword.keyword_id, keyword.keyword, AVG(keyword.id) FROM movie_link JOIN title ON title.id = movie_link.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id JOIN keyword ON keyword.id = movie_keyword.keyword_id WHERE title.md5sum > '95873621bb7f3d8f2fc2d7d21c8b9817' AND title.episode_of_id > 1325926 GROUP BY movie_link.movie_id, title.imdb_index, movie_keyword.keyword_id, keyword.keyword ORDER BY title.imdb_index ASC, keyword.keyword DESC;
