select complete_cast.movie_id, title.title, title.series_years, complete_cast.subject_id, keyword.phonetic_code from keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id where keyword.id >= 104859 AND keyword.keyword <= 'civic-pride' AND complete_cast.movie_id >= 1438895 order by title.title DESC, keyword.phonetic_code DESC;
