SELECT cast_info.id, role_type.role, char_name.name_pcode_nf, MIN(role_type.id) FROM role_type JOIN cast_info ON cast_info.role_id = role_type.id JOIN char_name ON char_name.id = cast_info.person_role_id WHERE cast_info.nr_order < 29 AND cast_info.person_id = 1260380 GROUP BY cast_info.id, role_type.role, char_name.name_pcode_nf HAVING MIN(role_type.id) = 10 ORDER BY MIN(role_type.id) ASC;
