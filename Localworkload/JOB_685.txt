SELECT title.season_nr, kind_type.id, aka_title.production_year, movie_companies.id, company_type.id FROM aka_title JOIN kind_type ON kind_type.id = aka_title.kind_id JOIN title ON title.kind_id = kind_type.id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE aka_title.movie_id = 1853323 AND movie_companies.id >= 1721230 ORDER BY movie_companies.id ASC, title.season_nr DESC, company_type.id DESC, aka_title.production_year ASC;
