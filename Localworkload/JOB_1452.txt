SELECT title.season_nr, movie_link.movie_id, movie_companies.note, link_type.link, company_type.id FROM link_type JOIN movie_link ON movie_link.link_type_id = link_type.id JOIN title ON title.id = movie_link.movie_id JOIN movie_companies ON movie_companies.movie_id = title.id JOIN company_type ON company_type.id = movie_companies.company_type_id WHERE title.episode_of_id >= 384768 AND movie_companies.note >= '(1958) (USA) (TV)' ORDER BY title.season_nr ASC, movie_companies.note ASC, link_type.link DESC;
