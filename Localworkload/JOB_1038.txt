SELECT title.production_year, movie_link.link_type_id, company_type.id, movie_companies.company_type_id FROM company_type JOIN movie_companies ON movie_companies.company_type_id = company_type.id JOIN title ON title.id = movie_companies.movie_id JOIN movie_link ON movie_link.movie_id = title.id WHERE title.season_nr = 5 AND title.md5sum <= 'a4ac763a4f937c3a819617091f8bcfab' AND movie_companies.company_id >= 3117 ORDER BY title.production_year DESC, company_type.id ASC;
