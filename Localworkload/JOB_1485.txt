select movie_keyword.movie_id, keyword.phonetic_code from keyword, movie_keyword where movie_keyword.keyword_id >= 33929 AND keyword.id = 86959 AND movie_keyword.movie_id >= 1737129 AND movie_keyword.id > 735536 AND keyword.phonetic_code < 'O1232' order by movie_keyword.movie_id DESC, keyword.phonetic_code DESC;
