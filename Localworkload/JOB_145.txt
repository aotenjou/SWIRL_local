SELECT title.md5sum, aka_title.movie_id, kind_type.kind, movie_keyword.movie_id, keyword.keyword FROM keyword JOIN movie_keyword ON movie_keyword.keyword_id = keyword.id JOIN title ON title.id = movie_keyword.movie_id JOIN aka_title ON aka_title.movie_id = title.id JOIN kind_type ON kind_type.id = aka_title.kind_id WHERE aka_title.note <= '(International: English title) (imdb display title)' AND title.id < 563730 ORDER BY aka_title.movie_id DESC, keyword.keyword ASC, movie_keyword.movie_id DESC, kind_type.kind DESC;
