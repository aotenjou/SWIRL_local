select comp_cast_type.id, movie_keyword.movie_id, movie_keyword.keyword_id, title.phonetic_code from movie_keyword JOIN title ON title.id = movie_keyword.movie_id JOIN complete_cast ON complete_cast.movie_id = title.id JOIN comp_cast_type ON comp_cast_type.id = complete_cast.status_id where title.series_years >= '2005-2009' AND comp_cast_type.kind > 'crew' AND title.episode_nr <= 29;
