select title.imdb_index, movie_info_idx.info, title.phonetic_code from movie_info_idx JOIN title ON title.id = movie_info_idx.movie_id JOIN kind_type ON kind_type.id = title.kind_id where kind_type.id = 1 AND movie_info_idx.id < 412529 AND title.kind_id <= 2 AND title.phonetic_code = 'D42' order by movie_info_idx.info DESC, title.phonetic_code ASC;
