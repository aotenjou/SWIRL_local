select title.series_years, movie_keyword.movie_id, cast_info.note from cast_info JOIN title ON title.id = cast_info.movie_id JOIN movie_keyword ON movie_keyword.movie_id = title.id where cast_info.person_role_id = 150668 AND cast_info.id <= 13329737 AND cast_info.role_id = 11 order by cast_info.note DESC;
